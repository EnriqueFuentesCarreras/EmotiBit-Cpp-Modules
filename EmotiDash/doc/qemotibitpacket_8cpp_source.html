147/**
148 * @brief Crea un paquete EmotiBit a partir de datos en formato QString.
149 *
150 * @param typeTag Tipo de paquete (ej: "EA", "HR", "AX").
151 * @param packetNumber Número de secuencia del paquete.
152 * @param data Datos del payload (separados por comas).
153 * @param numElements Número de elementos en el payload.
154 * @param protocolVersion Versión del protocolo utilizado.
155 * @param dataReliability Nivel de confiabilidad de los datos.
156 * @return QString Representación textual del paquete completo.
157 */
158QString qEmotiBitPacket::createPacket(const QString &typeTag, quint16 packetNumber, const QString &data, quint16 numElements, quint8 protocolVersion, quint8 dataReliability) {
159 // Crear el encabezado
160 qEmotiBitPacket::Header header;
161 header.typeTag = typeTag;
162 header.timestamp = QDateTime::currentMSecsSinceEpoch();
163 header.packetNumber = packetNumber;
164 header.dataLength = numElements; // Cambiar aquí para reflejar el número de elementos
165 header.protocolVersion = protocolVersion;
166 header.dataReliability = dataReliability;
167
168 // Construir el paquete como una cadena
169 QStringList packetElements;
170 packetElements << QString::number(header.timestamp)
171 << QString::number(header.packetNumber)
172 << QString::number(header.dataLength) // Usar el número de elementos
173 << header.typeTag
174 << QString::number(header.protocolVersion)
175 << QString::number(header.dataReliability);
176
177 // Añadir los datos si existen
178 if (!data.isEmpty()) {
179 packetElements << data.split(PAYLOAD_DELIMITER);
180 }
181
182 // Unir todos los elementos con el delimitador
183 QString packet = packetElements.join(PAYLOAD_DELIMITER);
184
185 // Añadir el delimitador de paquete
186 packet += PACKET_DELIMITER_CSV;
187
188 return packet;
189}

217/**
218 * @brief Extrae el encabezado de un paquete a partir de su representación en texto completo.
219 *
220 * @param packet Cadena completa del paquete recibido.
221 * @param packetHeader Estructura donde se almacenarán los datos extraídos del encabezado.
222 * @return qint16 Posición inicial donde comienzan los datos (payload) o MALFORMED_HEADER si hay error.
223 */
224qint16 qEmotiBitPacket::getHeader(const QString &packet, Header &packetHeader)
225{
226 qint16 dataStartChar = 0;
227 qint16 commaN = 0, commaN1;
228
229 // timestamp
230 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
231 //qDebug() << "1 dataStartChar commaN1 " << commaN1;
232 if (commaN1 == -1 || commaN1 <= commaN) return MALFORMED_HEADER;
233 packetHeader.timestamp = packet.mid(commaN, commaN1 - commaN).toLongLong();
234
235 // packetNumber
236 commaN = commaN1 + 1;
237 //qDebug() << "2 dataStartChar commaN1 " << commaN1;
238 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
239 if (commaN1 == -1 || commaN1 <= commaN) return MALFORMED_HEADER;
240 packetHeader.packetNumber = packet.mid(commaN, commaN1 - commaN).toInt();
241
242 // dataLength
243 commaN = commaN1 + 1;
244 //qDebug() << "3 dataStartChar commaN1 " << commaN1;
245 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
246 if (commaN1 == -1 || commaN1 <= commaN) return MALFORMED_HEADER;
247 packetHeader.dataLength = packet.mid(commaN, commaN1 - commaN).toInt();
248
249 // typeTag
250 commaN = commaN1 + 1;
251 //qDebug() << "4 dataStartChar commaN1 " << commaN1;
252 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
253 if (commaN1 == -1 || commaN1 <= commaN) return MALFORMED_HEADER;
254 packetHeader.typeTag = packet.mid(commaN, commaN1 - commaN);
255
256 // protocolVersion
257 commaN = commaN1 + 1;
258 //qDebug() << "5 dataStartChar commaN1 " << commaN1;
259 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
260 //qDebug() << "6 dataStartChar commaN1 " << commaN1;
261 if (commaN1 == -1 || commaN1 <= commaN) return MALFORMED_HEADER;
262 packetHeader.protocolVersion = packet.mid(commaN, commaN1 - commaN).toInt();
263
264 // dataReliability
265 commaN = commaN1 + 1;
266 //qDebug() << "7 dataStartChar commaN1 " << commaN1;
267 commaN1 = packet.indexOf(PAYLOAD_DELIMITER, commaN);
268 //qDebug() << "8 dataStartChar commaN1 " << commaN1;
269 if (commaN1 == -1) {
270 // No hay más delimitadores; estamos al final del paquete
271 commaN1 = packet.size();
272 //qDebug() << "9 dataStartChar commaN1 " << commaN1;
273 dataStartChar = NO_PACKET_DATA; // No hay datos adicionales
274 } else {
275 dataStartChar = commaN1 + 1;
276 //qDebug() << "1 dataStartChar " << dataStartChar;
277 }
278
279 if (commaN1 <= commaN) return MALFORMED_HEADER;
280 packetHeader.dataReliability = packet.mid(commaN, commaN1 - commaN).toInt();
281 // qDebug() << "PAQUETE DataStartChar " << dataStartChar;
282 return dataStartChar;
283}

293/**
294 * @brief Extrae el encabezado de un paquete a partir de su representación en texto completo.
295 *
296 * @param packet Cadena completa del paquete recibido.
297 * @param packetHeader Estructura donde se almacenarán los datos extraídos del encabezado.
298 * @return qint16 Posición inicial donde comienzan los datos (payload) o MALFORMED_HEADER si hay error.
299 */
300bool qEmotiBitPacket::getHeader(const QStringList &packetElements, Header &packetHeader) {
301 if (packetElements.size() < HEADER_LENGTH) {
302 return false; // El paquete es demasiado corto
303 }
304
305 bool ok;
306 packetHeader.timestamp = packetElements[0].toULongLong(&ok);
307 if (!ok) return false;
308
309 packetHeader.packetNumber = packetElements[1].toUShort(&ok);
310 if (!ok) return false;
311
312 packetHeader.dataLength = packetElements[2].toUShort(&ok);
313 if (!ok) return false;
314
315 packetHeader.typeTag = packetElements[3];
316
317 packetHeader.protocolVersion = packetElements[4].toUShort(&ok);
318 if (!ok) return false;
319
320 packetHeader.dataReliability = packetElements[5].toUShort(&ok);
321 if (!ok) return false;
322
323 return true;
324}

328/**
329 * @brief Extrae un único elemento (campo) del paquete a partir de una posición dada.
330 *
331 * @param packet Paquete completo como cadena de texto.
332 * @param element Variable donde se almacenará el elemento extraído.
333 * @param startChar Posición inicial para comenzar la extracción.
334 * @return qint16 Nueva posición para la siguiente extracción o NO_PACKET_DATA si no hay más datos.
335 */
336qint16 qEmotiBitPacket::getPacketElement(const QString &packet, QString &element, qint16 startChar){
337 int nextStartChar = -1;
338 int commaN1 = packet.indexOf(PAYLOAD_DELIMITER, startChar);
339 // qDebug() << "getPacketElement(packet " << packet << ",element " << element << ", startChar " << startChar;
340 if (commaN1 != -1) {
341 // Se encontró una coma, extraer elemento
342 element = packet.mid(startChar, commaN1 - startChar);
343 if (packet.length() > commaN1 + 1) {
344 nextStartChar = commaN1 + 1;
345 }
346 }
347 else if (packet.length() > startChar) {
348 // No se encontró una coma, retornar el último elemento
349 element = packet.mid(startChar);
350 nextStartChar = NO_PACKET_DATA;
351 }
352 //qDebug() << "return nextStartChar " << nextStartChar;
353 return nextStartChar;
354}

359/**
360 * @brief Busca y extrae el valor asociado a una clave (key) en un paquete representado como QStringList.
361 *
362 * @param packet Lista de elementos separados del paquete.
363 * @param key Clave a buscar.
364 * @param value Variable donde se almacenará el valor asociado a la clave.
365 * @param startIndex Índice inicial donde comenzar la búsqueda.
366 * @return qint16 Índice donde se encontró el valor o -1 si no se encontró.
367 */
368qint16 qEmotiBitPacket::getPacketKeyedValue(const QStringList &packet, const QString &key, QString &value, int startIndex){
369 //qDebug() << "getPacketKeyedValue(packet " << packet << ",key " << key << ", value, startIndex " << startIndex;
370 for (int i = startIndex; i < packet.size(); i++) {
371 if (packet.at(i).compare(key, Qt::CaseInsensitive) == 0) {
372 if (i + 1 < packet.size()) {
373 value = packet.at(i + 1);
374 return i + 1;
375 }
376 return -1; // No se encontró un valor después de la clave
377 }
378 }
379 return -1; //
380}

384/**
385 * @brief Busca y extrae el valor asociado a una clave (key) en un paquete representado como QString.
386 *
387 * @param packet Paquete completo como cadena de texto.
388 * @param key Clave a buscar.
389 * @param value Variable donde se almacenará el valor asociado a la clave.
390 * @param startChar Posición inicial para comenzar la búsqueda.
391 * @return qint16 Nueva posición tras encontrar el valor o -1 si no se encontró.
392 */
393qint16 qEmotiBitPacket::getPacketKeyedValue(const QString &packet, const QString &key, QString &value, int startChar){
394 //qDebug() << "getPacketKeyedValue(packet " << packet << ",key " << key << ", value, startChar__ " << startChar;
395 QString element;
396 do {
397 startChar = qEmotiBitPacket::getPacketElement(packet, element, startChar);
398 if (element.compare(key, Qt::CaseInsensitive) == 0) {
399 //startChar = qEmotiBitPacket::getPacketElement(packet, value, startChar);
400 qEmotiBitPacket::getPacketElement(packet, value, startChar);
401 if (!value.isEmpty()) {
402 // qDebug() << "startChar " << startChar;
403 return startChar;
404 }
405 return -1; // Retorna -1 si se alcanza el final del paquete antes de encontrar un valor
406 }
407 } while (startChar > -1);
408 return -1; // Retorna -1 si se alcanza el final del paquete antes de encontrar la clave
409}
