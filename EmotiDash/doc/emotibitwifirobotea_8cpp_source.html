225/*
226 * \brief Inicia la conexión UDP para recepción de datos desde EmotiBit.
227 * Este método intenta enlazar el socket `dataCxn` a un puerto dado. Si falla,
228 * incrementa el puerto y vuelve a intentar hasta 10 veces. Si tiene éxito,
229 * configura el buffer de recepción y otras opciones del socket.
230 * \param dataPort Puerto inicial que se intenta usar para la conexión de datos.
231 * \return quint8
232 * - SUCCESS si la vinculación fue exitosa.
233 * - FAIL si no se pudo enlazar después de varios intentos.
234 */
235quint8 EmotiBitWiFiRoboTEA::_startDataCxn(quint16 dataPort){
236 //Intenta enlazar el socket UDP dataCxn al puerto de datos que se pasa, incrementando el
237 //puerto si falla hasta máximo de 10 intentos, configura el socket
238 //si la vinculación es exitosa y devuelve SUCCESS o FAIL.
239 _dataPort = dataPort;
240
241
242 // Intentar enlazar el socket al puerto especificado
243 bool bound = false;
244 const int MAX_RETRIES = 10;
245 int tries = 0;
246
247 while (!bound && tries < MAX_RETRIES) {
248 if (dataCxn->bind(QHostAddress::Any, _dataPort, QAbstractSocket::DefaultForPlatform)) {
249 qDebug() << "Enlazado el socket al puerto:" << _dataPort;
250
251 bound = true;
252 }
253 else {
254 // Incrementa el puerto y vuelve a intentar
255 _dataPort += 2;
256 qDebug() << "Trying data port:" << _dataPort;
257 tries++;
258 }
259 }
260
261 if (!bound) {
262 qDebug() << "Failed to bind dataCxn after multiple attempts.";
263 return FAIL;
264 }
265
266 // Configurar el socket_____________________________________________________________________________________
267 //dataCxn->setSocketOption(QAbstractSocket::ReceiveBufferSizeSocketOption, QVariant(32768)); // Tamaño del buffer: 2^15 bytes 131072
268 dataCxn->setSocketOption(QAbstractSocket::ReceiveBufferSizeSocketOption, QVariant(131072));
269 dataCxn->setSocketOption(QAbstractSocket::LowDelayOption, true); // No es necesario en UDP, pero se incluye para consistencia
270 dataCxn->setSocketOption(QAbstractSocket::MulticastTtlOption, QVariant(1)); // Ejemplo de opción adicional para multicast
271 //-------------------------------------------------------------------------------------------------------------
272
273 // Obtener información para depuración
274 QVariant bufferSizeVar = dataCxn->socketOption(QAbstractSocket::ReceiveBufferSizeSocketOption);
275 int receiveBufferSize = bufferSizeVar.toInt();
276 qDebug() << "dataCxn GetReceiveBufferSize:" << receiveBufferSize;
277 qDebug() << "dataCxn initialized successfully on port:" << _dataPort;
278 return SUCCESS;
279}

337QVector<QString> EmotiBitWiFiRoboTEA::getLocalIPs() {
338 QVector<QString> result;
339 QProcess process;
340 process.start("ipconfig");
341 process.waitForFinished();
342 QString commandResult = process.readAllStandardOutput();
343
344 int pos = 0;
345 while ((pos = commandResult.indexOf("IPv4", pos)) >= 0) {
346 pos = commandResult.indexOf(":", pos) + 2;
347 int pos2 = commandResult.indexOf("\n", pos);
348
349 QString ip = commandResult.mid(pos, pos2 - pos).trimmed();
350 pos = pos2;
351
352 if (!ip.startsWith("127")) { // Omitir direcciones loopback
353 qDebug() << "la funcion getLocalIps() encontro: " << ip;
354 result.append(ip);

417/*
418 * \brief Comprueba si una dirección IP pertenece a alguna red de una lista dada.
419 *
420 * Compara la dirección `ipAddress` con cada entrada de `networkList`, aceptando
421 * comodines (`*`) en los segmentos. Devuelve true si hay coincidencia parcial o total.
422 *
423 * \param ipAddress Dirección IP a verificar.
424 * \param networkList Lista de redes en formato de string (ej: "192.168.1").
425 * \return true si la IP pertenece a alguna red de la lista, false si no.
426 */
427bool EmotiBitWiFiRoboTEA::isInNetworkList(const QString &ipAddress, const QList<QString> &networkList) const {
428 bool out = false;
429
430 // Dividir la dirección IP en partes
431 QStringList ipSplit = ipAddress.split('.');
432
433 // Recorrer la lista de redes
434 for (const QString &listIp : networkList) {
435 // Dividir la dirección de la lista en partes
436 QStringList listIpSplit = listIp.split('.');
437 bool partMatch = true;
438
439 // Comparar cada segmento de la dirección IP
440 for (int n = 0; n < ipSplit.size() && n < listIpSplit.size(); n++) {
441 if (listIpSplit.at(n) == "*" || listIpSplit.at(n) == ipSplit.at(n)) {
442 // Coincidencia parcial
443 continue;
444 } else {
445 partMatch = false;
446 break;
447 }
448 }
449 if (partMatch) {
450 // Se encontró una coincidencia
451 out = true;
452 break;
453 }
454 }
455 return out;
456}
