176/*
177 * onNewDataReceived
178 * Recibe nuevos datos y actualiza la serie y el buffer del canal correspondiente.
179 * También marca la gráfica como "sucia" para recalcular los ejes.
180 *
181 * @param channelID Identificador del canal (ej. "AX", "HR", etc.)
182 * @param t Tiempo en segundos
183 * @param y Valor del dato
184 */
185void FormPlot::onNewDataReceived(const QString &channelID,
186 double t,
187 double y)
188{
189 constexpr int MAX_SAMPLES = 1000;
190 constexpr double windowSize = 10.0; // segundos visibles
191 const double tMin = t - windowSize;
192
193 // --- serie y buffer ---
194 auto *series = seriesMap.value(channelID, nullptr);
195 if (!series) return;
196
197 auto &buf = buffers[channelID];
198 buf.data.append({t, y});
199 while (!buf.data.isEmpty() && buf.data.first().x() < tMin)
200 buf.data.removeFirst();
201 if (buf.data.size() > MAX_SAMPLES)
202 buf.data.remove(0, buf.data.size() - MAX_SAMPLES);
203
204 QVector<QPointF> v = buf.data;
205 if (!v.isEmpty() && t - v.last().x() > 0.05 * windowSize)
206 v.append({t, v.last().y()}); // punto fantasma
207 series->replace(v);
208
209 //--- eje X ---
210 auto &cache = plotCache[channel2plot.value(channelID)];
211 if (cache.axX) cache.axX->setRange(tMin, t);
212 cache.dirty = true; // recalcular Y
213
214 // --- etiqueta de tiempo (opcional) ---
215 if (ui->labelTime)
216 ui->labelTime->setText(QString::number(t, 'f', 2));
217}

228void FormPlot::refreshCharts(){
229 for (auto it = chartViewMap.cbegin(); it != chartViewMap.cend(); ++it) {
230 const QString plotName = it.key();
231 QChartView *view = it.value();
232 auto &cache = plotCache[plotName];
233
234 if (cache.dirty && cache.axY) {
235 const double xMin = cache.axX->min();
236 const double xMax = cache.axX->max();
237
238 double minY = std::numeric_limits<double>::max();
239 double maxY = -std::numeric_limits<double>::max();
240
241 for (auto *s : view->chart()->series()) {
242 auto *ls = qobject_cast<QLineSeries*>(s);
243 if (!ls) continue;
244 for (const QPointF &p : ls->pointsVector()) {
245 if (p.x() < xMin || p.x() > xMax) continue;
246 minY = std::min(minY, p.y());
247 maxY = std::max(maxY, p.y());
248 }
249 }
250
251 if (minY == std::numeric_limits<double>::max()) {
252 minY = -1.0; maxY = 1.0;
253 } else if (minY == maxY) {
254 minY -= 1.0; maxY += 1.0;
255 } else {
256 const double margin = 0.10 * (maxY - minY);
257 minY -= margin;
258 maxY += margin;
259 }
260
261 // Suavizado exponencial
262 const double alpha = 0.4; // más pequeño = más suavizado
263
264 if (cache.smoothMinY == cache.smoothMaxY) {
265 // Primera vez
266 cache.smoothMinY = minY;
267 cache.smoothMaxY = maxY;
268 } else {
269 cache.smoothMinY = (1.0 - alpha) * cache.smoothMinY + alpha * minY;
270 cache.smoothMaxY = (1.0 - alpha) * cache.smoothMaxY + alpha * maxY;
271 }
272 cache.axY->setRange(cache.smoothMinY, cache.smoothMaxY);
273
274 // Formato dinámico de etiquetas
275 double absMax = std::max(std::fabs(cache.smoothMinY), std::fabs(cache.smoothMaxY));
276 int decimals;
277 if (absMax < 100.0)
278 decimals = 4;
279 else if (absMax < 1000.0)
280 decimals = 3;
281 else if (absMax < 10000.0)
282 decimals = 1;
283 else
284 decimals = 0;
285 QString format = QString("%.%1f").arg(decimals);
286 cache.axY->setLabelFormat(format);
287 cache.dirty = false;
288 }
289 if (view) view->update(); // repintar vista
290 }
291}

342/*
343 * advanceTimeWindow
344 * Avanza suavemente la ventana de tiempo mostrada en el eje X de cada gráfica.
345 * Realiza scroll en píxeles y sincroniza los rangos cada cierto tiempo.
346 */
347void FormPlot::advanceTimeWindow()
348{
349 // tiempo actual
350 const double now = QDateTime::currentMSecsSinceEpoch()/1000.0;
351
352 // rawDt
353 const double rawDt = now - currentTime;
354 currentTime = now;
355 if (rawDt <= 0.0) return;
356
357 // suavizado exponencial del delta-tiempo
358 constexpr double alpha = 0.2;
359 if (smoothDt == 0.0) // primer frame
360 smoothDt = rawDt;
361 else
362 smoothDt = alpha*rawDt + (1.0 - alpha)*smoothDt;
363 const double dt = smoothDt;
364
365 // actualiza etiqueta de tiempo (opcional)
366 if (ui->labelTime)
367 ui->labelTime->setText(QString::number(now, 'f', 2));
368
369 // desplazamiento en cada gráfico
370 // calculamos píxeles/segundo una sola vez por frame
371 double wPix = chartViewMap.isEmpty() ? 0.0
372 : chartViewMap.begin().value()->viewport()->width();
373 const double pxPerSec = wPix / windowSize;
374
375 for (auto it = plotCache.begin(); it != plotCache.end(); ++it)
376 {
377 const QString &plotName = it.key();
378 PlotCache &cache = it.value();
379 QChartView *view = chartViewMap.value(plotName, nullptr);
380 if (!view || !cache.axX) continue;
381
382 // ) scroll "suave" con fracción acumulada
383 double moveAcc = pxPerSec * dt + cache.fracPixPend;
384 int dxPix = int(moveAcc);
385 cache.fracPixPend = moveAcc - dxPix;
386 if (dxPix > 0)
387 view->chart()->scroll(dxPix, 0);
388
389 // cada ~0.5 s sincronizamos el rango X
390 if (now - cache.lastT > 0.5) {
391 cache.axX->setRange(now - windowSize, now);
392 cache.lastT = now;
393 }
394 }
395}
