3 * @brief Interfaz gráfica principal para la interacción con un dispositivo EmotiBit.

5 * Esta clase proporciona un formulario Qt que permite:

193/**
194 * @brief Inicia la búsqueda de dispositivos EmotiBit disponibles en la red.
195 *
196 * Actualiza el ComboBox con los dispositivos detectados y su disponibilidad.
197 */
198void FormVistaEmotiBit::detectarPulsera(){
199 ui->textBrowserMensajes->append("Buscando dispositivos EmotiBit...");
200 ui->comboBoxDispositivos->clear();;
201
202 // Esperamos 2 seg y luego consultamos
203 QTimer::singleShot(2000, this, [this]() {
204 auto devices = controller.getDiscoveredDevices();
205
206 if (devices.empty()) {
207 ui->textBrowserMensajes->append("No se encontraron dispositivos EmotiBit.");
208 ui->pushButtonConectar->setEnabled(false);
209 } else {
210 for (const auto &[id, info] : devices) {
211 QString qId = QString::fromStdString(id);
212 if (info.isAvailable) {
213 QString deviceInfo = QString("%1 (%2)").arg(qId).arg(info.ip);
214 ui->comboBoxDispositivos->addItem(deviceInfo);
215 ui->textBrowserMensajes->append(
216 QString("Dispositivo detectado: <span style='color:blue;'>%1</span>, IP: %2, Estado <span style='color:green;'>Disponible</span>")
217 .arg(qId).arg(info.ip)
218 );
219 ui->pushButtonConectar->setEnabled(true);
220 } else {
221 ui->textBrowserMensajes->append(
222 QString("Dispositivo detectado:<span style='color:grey;'> %1</span>, IP: %2,<span style='color:red;'> NO Disponible</span>")
223 .arg(qId).arg(info.ip) // std
224 );
225 }
226 }
227 }
228 });
229}//_______________

234 * @brief Intenta conectar al dispositivo EmotiBit seleccionado en el ComboBox.
235 *
236 * Actualiza el estado de los botones y emite señales de conexión.
237 */
238void FormVistaEmotiBit::conectarPulsera(){
239 ui->textBrowserMensajes->append("***************** Intentando conectar al dispositivo EmotiBit...");
240 int selectedIndex = ui->comboBoxDispositivos->currentIndex();
241 if (selectedIndex == -1) {
242 ui->textBrowserMensajes->append("<span style='color:red;'>Error: No se ha seleccionado ningún dispositivo.</span>");
243 return;
244 }
245 // Extraemos el ID verdadero, asumiendo que en el combobox está "ID (x.x.x.x)"
246 QString selectedText = ui->comboBoxDispositivos->currentText();
247 QString selectedDeviceId = selectedText.split(' ').first();
248 // (Si deseas parsear IP u otra cosa, ajusta aquí)
249
250 bool success = controller.connectToDevice(selectedDeviceId);
251 if (success) {
252 ui->textBrowserMensajes->append("Nombre del dispositivo conectado: " + selectedDeviceId);
253 emit signalLogEmotiBit("Nombre del dispositivo conectado: " + selectedDeviceId);
254 ui->textBrowserMensajes->append("<span style='color:green;'>Connect:</span> conexión establecida con éxito.");
255 emit signalLogEmotiBit("<span style='color:green;'>Connect:</span> conexión establecida con éxito.");
256 ui->pushButtonDesconectar->setEnabled(true);
257 ui->pushButtonConectar->setEnabled(false);
258 emit signalIDEmotiBit(selectedDeviceId);
259 // Reiniciamos gráfica
260 if (formPlot) {
261 formPlot->resetAllData();
262 }
263
264 } else {
265 ui->textBrowserMensajes->append("<span style='color:red;'>Error al conectar con el dispositivo.</span>");
266 emit signalLogEmotiBit("<span style='color:red;'>Error al conectar con el dispositivo.</span>");
267 ui->pushButtonDesconectar->setEnabled(false);
268 }
269 emit signalOnlineEmotiBit(success);
270}//_______________________
