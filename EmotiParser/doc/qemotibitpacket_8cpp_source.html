104QString qEmotiBitPacket::createPacket(const QString &typeTag, quint16 packetNumber, const QString &data, quint16 numElements, quint8 protocolVersion, quint8 dataReliability) {
105 // Crear el encabezado
106 qEmotiBitPacket::Header header;
107 header.typeTag = typeTag;
108 header.timestamp = QDateTime::currentMSecsSinceEpoch();
109 header.packetNumber = packetNumber;
110 header.dataLength = numElements; // Cambiar aquí para reflejar el número de elementos
111 header.protocolVersion = protocolVersion;
112 header.dataReliability = dataReliability;
113
114 // Construir el paquete como una cadena
115 QStringList packetElements;
116 packetElements << QString::number(header.timestamp)
117 << QString::number(header.packetNumber)
118 << QString::number(header.dataLength) // Usar el número de elementos
119 << header.typeTag
120 << QString::number(header.protocolVersion)
121 << QString::number(header.dataReliability);
122
123 // Añadir los datos si existen
124 if (!data.isEmpty()) {
125 packetElements << data.split(PAYLOAD_DELIMITER);
126 }
127
128 // Unir todos los elementos con el delimitador
129 QString packet = packetElements.join(PAYLOAD_DELIMITER);
130
131 // Añadir el delimitador de paquete
132 packet += PACKET_DELIMITER_CSV;
133
134 return packet;
135}

218bool qEmotiBitPacket::getHeader(const QStringList &packetElements, Header &packetHeader) {
219 if (packetElements.size() < HEADER_LENGTH) {
220 return false; // El paquete es demasiado corto
221 }
222
223 bool ok;
224 packetHeader.timestamp = packetElements[0].toULongLong(&ok);
225 if (!ok) return false;
226
227 packetHeader.packetNumber = packetElements[1].toUShort(&ok);
228 if (!ok) return false;
229
230 packetHeader.dataLength = packetElements[2].toUShort(&ok);
231 if (!ok) return false;
232
233 packetHeader.typeTag = packetElements[3];
234
235 packetHeader.protocolVersion = packetElements[4].toUShort(&ok);
236 if (!ok) return false;
237
238 packetHeader.dataReliability = packetElements[5].toUShort(&ok);
239 if (!ok) return false;
240
241 return true;
242}
